94 @ApiResponses(value = {
95 @ApiResponse(
96 responseCode = "200",

98 content = {
99 @Content(
100 mediaType = "application/json",
101 schema = @Schema(implementation = HostMeta.class)
102 )
103 }
104 ),
105 @ApiResponse(
106 responseCode = "400",
107 description = "Host meta does not exist with the given host_meta_id OR IP and/or Interface is missing",
108 content = @Content
109 ),
110 @ApiResponse(
111 responseCode = "500",
112 description = "Internal server error, contact admin",
113 content = @Content
114 )
115 })
116 public ResponseEntity<?> getHostMeta(@PathVariable("id") int id, @RequestParam(required = false) Integer version) {
117 try {
118 List<HostMeta> hm = hostMetaMapper.getHostMetaById(id, version);
119 return new ResponseEntity<>(hm, HttpStatus.OK);
120 }
121 catch (Exception ex) {
122 JSONObject jsonErr = new JSONObject();
123 jsonErr.put("id", id);
124 final Throwable cause = ex.getCause();
125 if (cause instanceof SQLException) {
126 LOGGER.error((cause).getMessage());
127 String state = ((SQLException) cause).getSQLState();
128 if (state.equals("45000")) {
129 jsonErr.put("message", "Record does not exist with the given host_meta_id");
130 }
131 else if (state.equals("45100")) {
132 jsonErr.put("message", "IP and/or Interface is missing");
133 }
134 return new ResponseEntity<>(jsonErr.toString(), HttpStatus.BAD_REQUEST);
135 }
136 return new ResponseEntity<>("Unexpected error", HttpStatus.INTERNAL_SERVER_ERROR);
137 }
