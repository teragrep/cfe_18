85 @RequestMapping(path = "", method = RequestMethod.PUT, produces = MediaType.APPLICATION_JSON_VALUE)
86 @Operation(summary = "Create flow")
87 @ApiResponses(value = {
88 @ApiResponse(responseCode = "201", description = "New flow created",
89 content = {@Content(mediaType = "application/json",
90 schema = @Schema(implementation = Flow.class))}),
91 @ApiResponse(responseCode = "400", description = "Internal SQL Exception",
92 content = @Content),
93 @ApiResponse(responseCode = "500", description = "Internal server error, contact admin", content = @Content)
94 })
95 public ResponseEntity<String> create(@RequestBody Flow newFlow) {
96 LOGGER.info("About to insert <[{}]>", newFlow);
97 try {
98 Flow f = flowMapper.create(newFlow.getName());
99 LOGGER.debug("Values returned <[{}]>", f);
100 JSONObject jsonObject = new JSONObject();
101 jsonObject.put("id", f.getId());
102 jsonObject.put("message", "New flow created");
103 return new ResponseEntity<>(jsonObject.toString(), HttpStatus.CREATED);
104 } catch (RuntimeException ex) {
105 LOGGER.error(ex.getMessage());
106 JSONObject jsonErr = new JSONObject();
107 jsonErr.put("id", newFlow.getId());
108 jsonErr.put("message", ex.getCause().toString());
109 return new ResponseEntity<>(jsonErr.toString(), HttpStatus.BAD_REQUEST);
110 }
111 }

113 @RequestMapping(path = "", method = RequestMethod.GET, produces = "application/json")
114 @Operation(summary = "Fetch all flows", description = "Will return empty list if there are no flows to fetch")

124 @RequestMapping(path = "/{id}", method = RequestMethod.DELETE, produces = MediaType.APPLICATION_JSON_VALUE)
125 @Operation(summary = "Delete flow")
126 @ApiResponses(value = {
127 @ApiResponse(responseCode = "200", description = "Flow deleted",
128 content = {@Content(mediaType = "application/json",
129 schema = @Schema(implementation = Flow.class))}),
130 @ApiResponse(responseCode = "400", description = "Flow is being used",
131 content = @Content),
132 @ApiResponse(responseCode = "404", description = "Flow does not exist",
133 content = @Content),
134 @ApiResponse(responseCode = "500", description = "Internal server error, contact admin", content = @Content)})
135 public ResponseEntity<String> delete(@PathVariable("id") int id) {
136 LOGGER.info("Deleting flow <[{}]>", id);
137 try {
138 flowMapper.delete(id);
139 JSONObject j = new JSONObject();
140 j.put("id", id);
141 j.put("message", "Flow deleted");
142 return new ResponseEntity<>(j.toString(), HttpStatus.OK);
143 } catch (RuntimeException ex) {
144 LOGGER.error(ex.getMessage());
145 JSONObject jsonErr = new JSONObject();
146 jsonErr.put("id", id);
147 jsonErr.put("message", ex.getCause().getMessage());
148 final Throwable cause = ex.getCause();
149 if (cause instanceof SQLException) {
150 String state = ((SQLException) cause).getSQLState();
151 if (state.equals("23000")) {
152 jsonErr.put("message", "Is in use");
153 return new ResponseEntity<>(jsonErr.toString(), HttpStatus.BAD_REQUEST);
154 } else if (state.equals("45000")) {
155 jsonErr.put("message", "Record does not exist");
156 return new ResponseEntity<>(jsonErr.toString(), HttpStatus.NOT_FOUND);
157 }
158 }
159 return new ResponseEntity<>(jsonErr.toString(), HttpStatus.BAD_REQUEST);
160 }
161 }
