87 @RequestMapping(path = "", method = RequestMethod.PUT, produces = "application/json")
88 @Operation(summary = "Create file based host")
89 @ApiResponses(value = {
90 @ApiResponse(responseCode = "201", description = "New file based host created",
91 content = {@Content(mediaType = "application/json",
92 schema = @Schema(implementation = HostFile.class))}),
93 @ApiResponse(responseCode = "400", description = "ID,MD5 or fqhost already exists",
94 content = @Content),
95 @ApiResponse(responseCode = "400", description = "Host exists for a different type or hub does not exist",
96 content = @Content),
97 @ApiResponse(responseCode = "500", description = "Internal server error, contact admin", content = @Content)
98 })
99 public ResponseEntity<String> create(@RequestBody HostFile newHostfile) {
100 LOGGER.info("About to insert <[{}]>", newHostfile);
101 try {
102 HostFile hf = hostFileMapper.create(
103 newHostfile.getMd5(),
104 newHostfile.getFqHost(),
105 newHostfile.getHubFq());
106 LOGGER.debug("Values returned <[{}]>", hf);
107 JSONObject jsonObject = new JSONObject();
108 jsonObject.put("id", hf.getId());
109 jsonObject.put("message", "New host created");
110 return new ResponseEntity<>(jsonObject.toString(), HttpStatus.CREATED);
111 } catch (RuntimeException ex) {
112 JSONObject jsonErr = new JSONObject();
113 jsonErr.put("id", newHostfile.getId());
114 jsonErr.put("message", ex.getCause().getMessage());
115 final Throwable cause = ex.getCause();
116 if (cause instanceof SQLException) {
117 String state = ((SQLException) cause).getSQLState();
118 if (state.equals("23000")) {
119 jsonErr.put("message", "ID,MD5 or fqhost already exists");
120 return new ResponseEntity<>(jsonErr.toString(), HttpStatus.BAD_REQUEST);
121 } else if (state.equals("45000")) {
122 jsonErr.put("message", "Host exists for a different type or hub does not exist");
123 return new ResponseEntity<>(jsonErr.toString(), HttpStatus.BAD_REQUEST);
124 }
125 }
126 return new ResponseEntity<>(jsonErr.toString(), HttpStatus.BAD_REQUEST);
127 }
128 }
