200 public void testHostMetaRetrieve() throws Exception {
201 List<String> IpList = new ArrayList<>();

204 stmnt.setInt(1, 1);

207 while (rs.next()) {

209 InterfaceList.add(rs.getString("interface"));
210 Assertions.assertEquals(1, rs.getInt("host_meta_id"));
211 Assertions.assertEquals("arch1", rs.getString("arch"));
212 Assertions.assertEquals("release_version1", rs.getString("release_version"));
213 Assertions.assertEquals("flavor1", rs.getString("flavor"));
214 Assertions.assertEquals("Linux1", rs.getString("os"));
215 Assertions.assertEquals("host1", rs.getString("hostname"));
216 }
217 Assertions.assertEquals(Arrays.asList("ip1", "ip1", "ip2", "ip2", "ip3", "ip3"), IpList);
218 Assertions
219 .assertEquals(Arrays.asList("ens192", "ens256", "ens192", "ens256", "ens192", "ens256"), InterfaceList);
