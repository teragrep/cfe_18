75 public void testProcessingTypeAcceptInsert() throws Exception {
76 // Gets the expected dataset
77 IDataSet expectedDataSet = new FlatXmlDataSetBuilder().build(new File("src/test/resources/XMLProcedureProcessingType/procedureProcessingTypeExpectedTestData1.xml"));
78 ITable expectedTable = expectedDataSet.getTable("cfe_18.file_processing_type");
79
80 CallableStatement stmnt = conn.prepareCall("{CALL cfe_18.insert_file_processing_type(?,?,?,?,?)}");
81
82 stmnt.setString(1, "TestTemplate1");
83 stmnt.setString(2, "TestRule1");
84 stmnt.setString(3, "TestName1");
85 stmnt.setString(4, "regex");
86 stmnt.setString(5, "TestRegex2");
87 stmnt.execute();
88
89 ITable actualTable = databaseConnection.createQueryTable("result", "select * from cfe_18.file_processing_type");
90
91 Assertion.assertEqualsIgnoreCols(expectedTable, actualTable, new String[]{"ruleset_id", "template_id"});

93 }
