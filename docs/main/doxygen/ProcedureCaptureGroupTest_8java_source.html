129 public void testRetrieveCaptureGroup() throws Exception {
130 List<Integer> actualList = new ArrayList<>();
131 CallableStatement stmnt = conn.prepareCall("{CALL cfe_18.retrieve_capture_group_details(?,?)}");
132 stmnt.setString(1, "capturegroup1");
133 stmnt.setString(2, null);
134 ResultSet rs = stmnt.executeQuery();
135 while (rs.next()) {
136 actualList.add(rs.getInt("capture_definition_id"));
137 String group_name = rs.getString("group_name");
138 String capture_type = rs.getString("capture_type");
139 int capture_group_id = rs.getInt("capture_group_id");
140 Assertions.assertEquals("capturegroup1", group_name);
141 Assertions.assertEquals("cfe", capture_type);
142 Assertions.assertEquals(1, capture_group_id);
143 }
144 Assertions.assertEquals(Arrays.asList(1, 2, 3, 4), actualList);
145
146 }
