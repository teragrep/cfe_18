89 @RequestMapping(path = "/meta/{id}", method = RequestMethod.GET, produces = "application/json")
90 @Operation(summary = "Fetch host meta by id")
91 @ApiResponses(value = {
92 @ApiResponse(responseCode = "200", description = "Host meta retrieved",
93 content = {@Content(mediaType = "application/json",
94 schema = @Schema(implementation = HostMeta.class))}),
95 @ApiResponse(responseCode = "400", description = "Host meta does not exist with the given host_meta_id OR IP and/or Interface is missing",
96 content = @Content),
97 @ApiResponse(responseCode = "500", description = "Internal server error, contact admin", content = @Content)
98 })
99 public ResponseEntity<?> getHostMeta(@PathVariable("id") int id, @RequestParam(required = false) Integer version) {
100 try {
101 List<HostMeta> hm = hostMetaMapper.getHostMetaById(id,version);
102 return new ResponseEntity<>(hm, HttpStatus.OK);
103 } catch (Exception ex) {
104 JSONObject jsonErr = new JSONObject();
105 jsonErr.put("id", id);
106 final Throwable cause = ex.getCause();
107 if (cause instanceof SQLException) {
108 LOGGER.error((cause).getMessage());
109 String state = ((SQLException) cause).getSQLState();
110 if (state.equals("45000")) {
111 jsonErr.put("message", "Record does not exist with the given host_meta_id");
112 } else if (state.equals("45100")) {
113 jsonErr.put("message", "IP and/or Interface is missing");
114 }
115 return new ResponseEntity<>(jsonErr.toString(), HttpStatus.BAD_REQUEST);
116 }
117 return new ResponseEntity<>("Unexpected error", HttpStatus.INTERNAL_SERVER_ERROR);
118 }
119 }

169 public ResponseEntity<String> addHostMeta(@RequestBody HostMeta newHostMeta) {

180 LOGGER.debug("Values returned <[{}]>",hm);
181 JSONObject jsonObject = new JSONObject();
